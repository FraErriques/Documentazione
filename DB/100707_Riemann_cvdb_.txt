declare @id int
select @id= 13
select 
	c.nominativo
	,s.nomeSettore
from 
	candidato c
	,settoreCandidatura_LOOKUP s
where
	c.id = @id
	and s.id = c.id_settore

create procedure usp_candidatoDescription_LOADSINGLE
	@id int
as
select 
	c.nominativo
	,s.nomeSettore
from 
	candidato c
	,settoreCandidatura_LOOKUP s
where
	c.id = @id
	and s.id = c.id_settore
GO
